 74 """""""""""""""""""" STATUSBAR
 75 
 76 hi StatColor guibg=#95e454 guifg=black ctermbg=lightgreen ctermfg=black
 77 hi Modified guibg=orange guifg=black ctermbg=lightred ctermfg=black
 78 hi PatchColor ctermfg=255 ctermbg=161 guifg=black guibg=#FF0066
 79 
 80 function! MyStatusLine(mode)
 81     let statusline=""
 82 
 83     if a:mode == 'Enter'
 84         let statusline.="%#StatColor#"
 85     endif
 86     let statusline.="\(%n\)\ %f\ "
 87     if a:mode == 'Enter'
 88         let statusline.="%*"
 89     endif
 90     let statusline.="%#Modified#%m"
 91     if a:mode == 'Leave'
 92         let statusline.="%*%r"
 93     elseif a:mode == 'Enter'
 94         let statusline.="%r%*"
 95     endif
 96     let statusline .= "\ (%l/%L,\ %c)\ %P%=%h%w\ %y\ [%{&encoding}:%{&fileformat}]\ \ "
 97     return statusline
 98 endfunction
 99 
100 au WinEnter * setlocal statusline=%!MyStatusLine('Enter')
101 au WinLeave * setlocal statusline=%!MyStatusLine('Leave')
102 set statusline=%!MyStatusLine('Enter')
103 
104 function! InsertStatuslineColor(mode)
105   if a:mode == 'i'
106     hi StatColor guibg=orange ctermbg=lightred
107   elseif a:mode == 'r'
108     hi StatColor guibg=#e454ba ctermbg=magenta
109   elseif a:mode == 'v'
110     hi StatColor guibg=#e454ba ctermbg=magenta
111   else
112     hi StatColor guibg=red ctermbg=red
113   endif
114 endfunction
115 
116 au InsertEnter * call InsertStatuslineColor(v:insertmode)
117 au InsertLeave * hi StatColor guibg=#95e454 guifg=black ctermbg=lightgreen ctermfg=black
